def get_stock_hist_data(code):
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 6.1; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/80.0.3987.149 Safari/537.36',
        'Referer': 'http://quote.eastmoney.com/.html'
    }
    params = (
        ('secid', '1.%s'%code if code[0] in ['6'] else '0.%s'%code),
        ('ut', 'fa5fd1943c7b386f172d6893dbfba10b'),
        ('fields1', 'f1,f2,f3,f4,f5'),
        ('fields2', 'f51,f52,f53,f54,f55,f56,f57,f61'),
        ('klt', '101'),
        ('fqt', '0'),
        ('beg', '0'),
        ('end', '20250101'),
#         ('lmt', '10000'),
        ('_', '1585797862454'),
    )
    res = requests.get('http://72.push2his.eastmoney.com/api/qt/stock/kline/get', headers=headers, params=params, verify=False)
    data = res.json()['data']
    return data


def clean_df(data,code):
    klines_list = [i.split(',') for i in data.get('klines')]
    df = pd.DataFrame(klines_list, columns=['date','open','close','high','low','vol','amount','turnover'])
    df['close'] = df['close'].astype(float)
    df['open'] = df['open'].astype(float)
    df['code'] = data['code']
    df['name'] = data['name']
    return df

def split_train_and_test(df):
    y = df['close'].shift(-1)
    X = df[['close']]
    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=66,shuffle=False)
    return X_train, X_test, y_train, y_test
